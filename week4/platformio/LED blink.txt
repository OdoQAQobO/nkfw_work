#include <Arduino.h>

int led_pin = 23;
int brightness = 0;
int fade_amount = 2;
int freq = 5000;
int led_channel = 0;
int resolution = 8; 

void setup() {
  ledcSetup(led_channel, freq, resolution);
  ledcAttachPin(led_pin, led_channel);
}

void loop() {
  ledcWrite(led_channel,brightness);
  brightness += fade_amount;
  
  if(brightness <= 0 || brightness >= 255 ) {
    fade_amount = -fade_amount; 
  }
  delay(30);
}
